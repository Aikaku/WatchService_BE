.
├── ai/                               # AI 분석 도메인 (AI 서버 연동 및 응답 처리)
│   ├── AiController.java              # [Controller] AI 관련 요청 처리 (예: 분석 요청 API)
│   ├── AiService.java                 # [Service] AI 서버와 통신, 분석 요청 및 결과 수신 처리
│   ├── domain/
│   │   └── AiResult.java              # [Domain Model] AI가 반환한 분석 결과를 표현 (예: 위험도, 판정 결과 등)
│   └── dto/
│       ├── AiPayload.java             # [DTO] AI 서버로 요청을 보낼 때 사용하는 데이터 포맷
│       └── AiResponse.java            # [DTO] AI 서버로부터 수신하는 응답 데이터 포맷

├── collector/                         # 파일 분석 도메인 (Watcher에서 전달된 파일을 실제로 분석)
│   ├── business/                      # Collector의 핵심 비즈니스 로직 (도메인 서비스 계층)
│   │   ├── EntropyAnalyzer.java       # [Domain Service] 파일의 Shannon 엔트로피를 계산 (무작위도 측정)
│   │   └── HashCalculator.java        # [Domain Service] 파일의 해시(MD5, SHA-256 등) 값을 계산
│   ├── domain/
│   │   └── FileData.java              # [Domain Model] 분석 대상 파일의 속성(경로, 크기, 수정 시간 등)을 표현
│   ├── dto/
│   │   └── FileAnalysisResult.java    # [DTO] Collector의 분석 결과 (파일 해시, 엔트로피 등)를 다른 도메인에 전달
│   └── FileCollectorService.java      # [Application Service] Collector의 상위 서비스.
│                                      # - FileData 수집 → Hash/Entropy 계산기 호출 → 결과(FileAnalysisResult) 생성 및 반환

├── common/                            # 공통 모듈 (도메인 간 공유되는 설정, 예외, 유틸리티)
│   ├── config/                        # [Config] Spring 설정 클래스, Bean 등록 등 환경 구성
│   ├── exception/                     # [Exception] 전역 예외 처리기 및 커스텀 예외 정의
│   └── util/                          # [Utility] 공통 유틸 함수 (파일, JSON 변환, 해시, 시간 관련 등)

├── storage/                           # 데이터 저장 도메인 (Collector의 결과를 DB 또는 로그 파일에 저장)
│   ├── domain/
│   │   └── Log.java                   # [Domain Model / Entity] 저장되는 로그 데이터 (파일 이벤트 기록)
│   ├── LogController.java             # [Controller] 로그 조회/관리용 API 제공
│   ├── LogRepository.java             # [Repository] SQLite 등 로컬 DB 접근, 로그 CRUD 구현
│   └── LogService.java                # [Service] 로그 저장, 조회, 삭제 등의 비즈니스 로직 수행

├── watcher/                           # 파일 감시 도메인 (사용자가 지정한 폴더 내 파일 변화를 감시)
│   ├── domain/
│   │   ├── WatcherEvent.java          # [Domain Model] 파일/폴더 감시 이벤트 (CREATE, MODIFY, DELETE 등)를 표현
│   │   └── WatcherPath.java           # [Domain Model] 감시 대상 경로를 관리 (경로 검증, 상대경로 계산 등)
│   ├── dto/
│   │   └── WatcherEventRecord.java    # [DTO] 감지된 이벤트를 Collector 도메인으로 전달하기 위한 데이터 객체
│   ├── WatcherController.java         # [Controller] 감시 시작/중지 등의 사용자 요청(API) 처리
│   ├── WatcherRepository.java         # [Repository] (옵션) 감시 이력, 설정 정보 등을 로컬 저장소에 보존할 때 사용
│   └── WatcherService.java            # [Service] OS WatchService 기반 파일 감시 로직 구현.
│                                      # - 폴더 등록/감시 실행/이벤트 감지 및 Collector로 전달까지 담당

└── WatchserviceAgentApplication.java  # [Spring Boot Entry Point] 스프링 애플리케이션 진입점.
                                       # 모든 도메인을 로드하고 서버 실행 (main 클래스)
